109     /**
110      * 
111      * @param configurationFile
112      *            XML format
113      * @param monitor
114      *            the associated monitor
115      * @param mib
116      *            the associated MIB
117      * 
118      * @throws IllegalArgumentException
119      */
120     public WaarpSnmpAgent(File configurationFile, WaarpInterfaceMonitor monitor,
121             WaarpInterfaceMib mib) throws IllegalArgumentException {
122         /**
123          * Creates a base agent with boot-counter, config file, and a
124          * CommandProcessor for processing SNMP requests.
125          * 
126          * Parameters:
127          * 
128          * These files does not exist and are not used but has to be specified.
129          * Read snmp4j docs for more info
130          * 
131          * "bootCounterFile" - a file with serialized boot-counter information
132          * (read/write). If the file does not exist it is created on shutdown of
133          * the agent.
134          * 
135          * "configFile" - a file with serialized configuration information
136          * (read/write). If the file does not exist it is created on shutdown of
137          * the agent.
138          * 
139          * "commandProcessor" - the CommandProcessor instance that handles the
140          * SNMP requests.
141          */
142         super(new File(configurationFile.getParentFile(), "dummyConf.agent"),
143                 new File(configurationFile.getParentFile(),
144                         "dummyBootCounter.agent"), new CommandProcessor(
145                         new OctetString(MPv3.createLocalEngineID())));
146         if (!SnmpConfiguration.setConfigurationFromXml(configurationFile)) {
147             throw new IllegalArgumentException("Cannot load configuration");
148         }
149         this.address = SnmpConfiguration.address;
150         this.nbThread = SnmpConfiguration.nbThread;
151         this.isFilterAccessEnabled = SnmpConfiguration.isFilterAccessEnabled;
152         this.useTrap = SnmpConfiguration.isUsingTrap;
153         this.setTrapLevel(SnmpConfiguration.trapLevel);
154         this.listUsmUser = SnmpConfiguration.listUsmUser;
155         this.listTargetElements = SnmpConfiguration.listTargetElements;
156         this.hasV2 = SnmpConfiguration.hasV2;
157         this.hasV3 = SnmpConfiguration.hasV3;
158 
159         logger.debug("SNMP Configuration loaded: " + this.address[0] + ":" +
160                 this.nbThread);
161         this.workerPool = ThreadPool.create("SnmpRequestPool", nbThread);
162         agent.setWorkerPool(this.workerPool);
163         this.setMonitor(monitor);
164         this.getMonitor().setAgent(this);
165         this.setMib(mib);
166         this.getMib().setAgent(this);
167     }

285     /**
286      * Adds initial notification targets and filters.
287      */
288     protected void addNotificationTargets(SnmpTargetMIB targetMIB,
289             SnmpNotificationMIB notificationMIB) {
290         targetMIB.addDefaultTDomains();
291 
292         for (TargetElement element : listTargetElements) {
293             logger.debug("AddTarget: " + element);
294             targetMIB.addTargetAddress(element.name, element.transportDomain,
295                     element.address, element.timeout, element.retries,
296                     element.tagList, element.params, element.storageType);
297         }
298         /**
299          * Example
300          * 
301          * targetMIB.addTargetAddress(new OctetString("notificationV2c"),
302          * TransportDomains.transportDomainUdpIpv4, new OctetString(new
303          * UdpAddress("127.0.0.1/162").getValue()), 200, 1, new
304          * OctetString("notify"), new OctetString("v2c"),
305          * StorageType.permanent); targetMIB.addTargetAddress(new
306          * OctetString("notificationV3"),
307          * TransportDomains.transportDomainUdpIpv4, new OctetString(new
308          * UdpAddress("127.0.0.1/1162").getValue()), 200, 1, new
309          * OctetString("notify"), new OctetString("v3notify"),
310          * StorageType.permanent);
311          */
312         logger.debug("HasV2: " + hasV2 + " HasV3: " + hasV3);
313         if (hasV2) {
314             targetMIB.addTargetParams(new OctetString(SnmpConfiguration.V2C),
315                     MessageProcessingModel.MPv2c,
316                     SecurityModel.SECURITY_MODEL_SNMPv2c, new OctetString(
317                             "cpublic"), SecurityLevel.AUTH_PRIV,
318                     StorageType.permanent);
319         }
320         if (hasV3) {
321             targetMIB.addTargetParams(new OctetString(
322                     SnmpConfiguration.V3NOTIFY), MessageProcessingModel.MPv3,
323                     SecurityModel.SECURITY_MODEL_USM, new OctetString(
324                             "v3notify"), SecurityLevel.NOAUTH_NOPRIV,
325                     StorageType.permanent);
326         }
327         int trapOrInform = SnmpNotificationMIB.SnmpNotifyTypeEnum.inform;
328         if (useTrap) {
329             trapOrInform = SnmpNotificationMIB.SnmpNotifyTypeEnum.trap;
330         }
331         notificationMIB.addNotifyEntry(new OctetString("default"),
332                 new OctetString(SnmpConfiguration.NOTIFY), trapOrInform,
333                 StorageType.permanent);
334     }

336     /**
337      * Minimal View based Access Control
338      * 
339      * http://www.faqs.org/rfcs/rfc2575.html
340      */
341     protected void addViews(VacmMIB vacm) {
342         vacm.addGroup(SecurityModel.SECURITY_MODEL_SNMPv1, new OctetString(
343                 "cpublic"), new OctetString("v1v2group"),
344                 StorageType.nonVolatile);
345         vacm.addGroup(SecurityModel.SECURITY_MODEL_SNMPv2c, new OctetString(
346                 "cpublic"), new OctetString("v1v2group"),
347                 StorageType.nonVolatile);
348         vacm.addGroup(SecurityModel.SECURITY_MODEL_USM, new OctetString(
349                 "v3notify"), new OctetString("v3group"),
350                 StorageType.nonVolatile);
351 
352         for (UsmUser user : listUsmUser) {
353             logger.debug("Groups: " + user.getSecurityName() + " Restricted? " +
354                     (user.getPrivacyProtocol() == null));
355             if (user.getPrivacyProtocol() == null) {
356                 vacm.addGroup(SecurityModel.SECURITY_MODEL_USM,
357                         new OctetString(user.getSecurityName()),
358                         new OctetString("v3restricted"),
359                         StorageType.nonVolatile);
360             } else {
361                 vacm.addGroup(SecurityModel.SECURITY_MODEL_USM,
362                         new OctetString(user.getSecurityName()),
363                         new OctetString("v3group"), StorageType.nonVolatile);
364             }
365         }
366 
367         vacm.addAccess(new OctetString("v1v2group"), new OctetString("public"),
368                 SecurityModel.SECURITY_MODEL_ANY, SecurityLevel.NOAUTH_NOPRIV,
369                 MutableVACM.VACM_MATCH_EXACT, new OctetString("fullReadView"),
370                 new OctetString("fullWriteView"), new OctetString(
371                         "fullNotifyView"), StorageType.nonVolatile);
372         vacm.addAccess(new OctetString("v3group"), new OctetString(),
373                 SecurityModel.SECURITY_MODEL_USM, SecurityLevel.AUTH_PRIV,
374                 MutableVACM.VACM_MATCH_EXACT, new OctetString("fullReadView"),
375                 new OctetString("fullWriteView"), new OctetString(
376                         "fullNotifyView"), StorageType.nonVolatile);
377         vacm.addAccess(new OctetString("v3restricted"), new OctetString(),
378                 SecurityModel.SECURITY_MODEL_USM, SecurityLevel.NOAUTH_NOPRIV,
379                 MutableVACM.VACM_MATCH_EXACT, new OctetString(
380                         "restrictedReadView"), new OctetString(
381                         "restrictedWriteView"), new OctetString(
382                         "restrictedNotifyView"), StorageType.nonVolatile);
383 
384         vacm.addViewTreeFamily(new OctetString("fullReadView"), new OID("1.3"),
385                 new OctetString(), VacmMIB.vacmViewIncluded,
386                 StorageType.nonVolatile);
387         vacm.addViewTreeFamily(new OctetString("fullWriteView"),
388                 new OID("1.3"), new OctetString(), VacmMIB.vacmViewIncluded,
389                 StorageType.nonVolatile);
390         vacm.addViewTreeFamily(new OctetString("fullNotifyView"),
391                 new OID("1.3"), new OctetString(), VacmMIB.vacmViewIncluded,
392                 StorageType.nonVolatile);
393 
394         vacm.addViewTreeFamily(new OctetString("restrictedReadView"), new OID(
395                 "1.3.6.1.2"), new OctetString(), VacmMIB.vacmViewIncluded,
396                 StorageType.nonVolatile);
397         vacm.addViewTreeFamily(new OctetString("restrictedWriteView"), new OID(
398                 "1.3.6.1.2.1"), new OctetString(), VacmMIB.vacmViewIncluded,
399                 StorageType.nonVolatile);
400         vacm.addViewTreeFamily(new OctetString("restrictedNotifyView"),
401                 new OID("1.3.6.1.2"), new OctetString(),
402                 VacmMIB.vacmViewIncluded, StorageType.nonVolatile);
403         vacm.addViewTreeFamily(new OctetString("restrictedNotifyView"),
404                 new OID("1.3.6.1.6.3.1"), new OctetString(),
405                 VacmMIB.vacmViewIncluded, StorageType.nonVolatile);
406     }

408     /**
409      * The table of community strings configured in the SNMP engine's Local
410      * Configuration Datastore (LCD).
411      * 
412      * We only configure one, "public".
413      */
414     protected void addCommunities(SnmpCommunityMIB communityMIB) {
415         Variable[] com2sec = new Variable[] {
416                 new OctetString("public"), // community name
417                 new OctetString("cpublic"), // security name
418                 getAgent().getContextEngineID(), // local engine ID
419                 new OctetString("public"), // default context name
420                 new OctetString(), // transport tag
421                 new Integer32(StorageType.nonVolatile), // storage type
422                 new Integer32(RowStatus.active) // row status
423         };
424         SnmpCommunityEntryRow row = communityMIB.getSnmpCommunityEntry().createRow(
425                 new OctetString("public2public").toSubIndex(true), com2sec);
426         communityMIB.getSnmpCommunityEntry().addRow(row);
427         if (isFilterAccessEnabled) {
428             snmpCommunityMIB.setSourceAddressFiltering(true);
429         }
430     }

432     @Override
433     protected void initTransportMappings() throws IOException {
434         TransportMapping<?>[] testMappings = new TransportMapping[address.length];
435         int nb = 0;
436         for (int i = 0; i < address.length; i++) {
437             Address addr = GenericAddress.parse(address[i]);
438             if (addr != null) {
439                 logger.info("SNMP Agent InitTransport: {} {}", addr.getClass()
440                         .getSimpleName(), addr);
441                 TransportMapping<?> tm = null;
442                 try {
443                     tm = TransportMappings.getInstance()
444                             .createTransportMapping(addr);
445                 } catch (RuntimeException e) {
446                     continue;
447                 }
448                 if (tm != null) {
449                     testMappings[nb] = tm;
450                     nb++;
451                 }
452             }
453         }
454         if (nb > 0) {
455             transportMappings = new TransportMapping[nb];
456             System.arraycopy(testMappings, 0, transportMappings, 0, nb);
457             /*for (int i = 0; i < nb; i ++) {
458                 transportMappings[i] = testMappings[i];
459             }*/
460             testMappings = null;
461         } else {
462             transportMappings = null;
463             throw new IOException("No address to connect");
464         }
465     }

467     /**
468      * Start method invokes some initialization methods needed to start the
469      * agent
470      * 
471      * @throws IOException
472      */
473     public void start() throws IOException {
474         logger.debug("WaarpSnmpAgent starting: " + address[0] + " 1 on " +
475                 address.length);
476         try {
477             init();
478         } catch (IOException e) {
479             logger.warn("Error while SNMP starts ", e);
480             throw e;
481         }
482         // This method reads some old config from a file and causes
483         // unexpected behavior.
484         // loadConfig(ImportModes.REPLACE_CREATE);
485         addShutdownHook();
486         getServer().addContext(new OctetString("public"));
487         finishInit();
488         run();
489         if (TrapLevel.StartStop.isLevelValid(getTrapLevel()))
490             sendColdStartNotification();
491     }

493     @Override
494     protected void sendColdStartNotification() {
495         logger.debug("ColdStartNotification: {}",
496                 getMib().getBaseOidStartOrShutdown());
497         SNMPv2MIB snmpv2 = this.getMib().getSNMPv2MIB();
498         notificationOriginator.notify(
499                 new OctetString("public"),
500                 SnmpConstants.coldStart,
501                 new VariableBinding[] {
502                         new VariableBinding(getMib().getBaseOidStartOrShutdown(),
503                                 new OctetString("Startup Service")),
504                         new VariableBinding(SnmpConstants.sysDescr, snmpv2
505                                 .getDescr()),
506                         new VariableBinding(SnmpConstants.sysObjectID, snmpv2
507                                 .getObjectID()),
508                         new VariableBinding(SnmpConstants.sysContact, snmpv2
509                                 .getContact()),
510                         new VariableBinding(SnmpConstants.sysName, snmpv2
511                                 .getName()),
512                         new VariableBinding(SnmpConstants.sysLocation, snmpv2
513                                 .getLocation()) });
514     }
515 
516     /**
517      * Send a Notification just before Shutdown the SNMP service.
518      */
519     protected void sendShutdownNotification() {
520         if (this.getMib() == null || notificationOriginator == null) {
521             return;
522         }
523         SNMPv2MIB snmpv2 = this.getMib().getSNMPv2MIB();
524         notificationOriginator.notify(
525                 new OctetString("public"),
526                 SnmpConstants.linkDown,
527                 new VariableBinding[] {
528                         new VariableBinding(getMib().getBaseOidStartOrShutdown(),
529                                 new OctetString("Shutdown Service")),
530                         new VariableBinding(SnmpConstants.sysDescr, snmpv2
531                                 .getDescr()),
532                         new VariableBinding(SnmpConstants.sysObjectID, snmpv2
533                                 .getObjectID()),
534                         new VariableBinding(SnmpConstants.sysContact, snmpv2
535                                 .getContact()),
536                         new VariableBinding(SnmpConstants.sysName, snmpv2
537                                 .getName()),
538                         new VariableBinding(SnmpConstants.sysLocation, snmpv2
539                                 .getLocation()) });
540         try {
541             Thread.sleep(100);
542         } catch (InterruptedException e) {
543         }
544     }
545 
546     @Override
547     public void stop() {
548         logger.info("Stopping SNMP support");
549         if (TrapLevel.StartStop.isLevelValid(getTrapLevel())) {
550             sendShutdownNotification();
551         }
552         super.stop();
553         if (getMonitor() != null) {
554             getMonitor().releaseResources();
555             try {
556                 Thread.sleep(100);
557             } catch (InterruptedException e) {
558             }
559             if (this.workerPool != null) {
560                 this.workerPool.cancel();
561             }
562         }
563     }
